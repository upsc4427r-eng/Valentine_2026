const yesBtn = document.getElementById("yes"); const noBtn = document.getElementById("no"); const question = document.getElementById("question"); const msg = document.getElementById("loveMsg"); const music = document.getElementById("music"); let popupVisible = false; let attempts = 0; // count chases not clicks const funnyTexts = [ "Nice try 😜", "Hehe can't catch me!", "Nope nope nope 😂", "Say YES already ❤️", "Too slow 😏", "Almost caught me 🤭" ]; /* MUSIC FIX FOR MOBILE */ document.body.addEventListener("click", () => music.play()); /* YES CLICK */ yesBtn.onclick = () => { question.style.display = "none"; yesBtn.style.display = "none"; noBtn.style.display = "none"; msg.style.display = "block"; confetti(); }; /* NO RUN AWAY + COUNT ATTEMPTS */ document.addEventListener("mousemove", (e) => { const rect = noBtn.getBoundingClientRect(); const dist = Math.hypot( e.clientX - (rect.left + rect.width/2), e.clientY - (rect.top + rect.height/2) ); if (dist < 130) { attempts++; // ⭐ count attempts moveNo(); showPopup(e.clientX, e.clientY); if (attempts >= 5) { attitudeMode(); attempts = 0; } } }); /* MOVE BUTTON */ function moveNo(){ const x = Math.random() * (window.innerWidth - 120); const y = Math.random() * (window.innerHeight - 70); noBtn.style.left = x + "px"; noBtn.style.top = y + "px"; } /* POPUP (one at a time) */ function showPopup(x,y){ if(popupVisible) return; popupVisible = true; const p = document.createElement("div"); p.className = "popup"; p.innerText = funnyTexts[Math.floor(Math.random()*funnyTexts.length)]; p.style.left = x + "px"; p.style.top = y + "px"; document.body.appendChild(p); setTimeout(()=>{ p.remove(); popupVisible=false; },1000); } /* YES DISAPPEAR MODE */ function attitudeMode(){ yesBtn.style.display = "none"; const a = document.createElement("div"); a.className = "attitude"; a.innerText = "Attitude huh !! 😏"; a.style.top = "40%"; a.style.left = "35%"; document.body.appendChild(a); setTimeout(()=>{ a.remove(); yesBtn.style.display = "block"; yesBtn.style.transform = "scale(1.4)"; setTimeout(()=> yesBtn.style.transform="scale(1)", 200); }, 2000); } /* CONFETTI */ function confetti(){ for(let i=0;i<70;i++){ const h=document.createElement("div"); h.className="heart"; h.innerHTML="💖"; h.style.left=Math.random()*100+"vw"; document.body.appendChild(h); } }
